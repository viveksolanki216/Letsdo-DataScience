load('data.leisure.Rda')
data.beauty <- data.leisure

Taking only recent data (2015) data

data.beauty2015 <- data.beauty[ data.beauty$start_date >= as.Date('2015-01-01') & 
                                  data.beauty$start_date <= as.Date('2015-09-30'),]

Get top deals taking sum of gb on life of deal only, not in deal bank

deal.gb <- data.beauty2015[ order_seq < deal_duration,list( gb = sum(gb)),by="deal_id"]
top_deals <- deal.gb[ deal.gb$gb >= quantile(deal.gb$gb,.7),]

Get top deals which have deal duration of 10 days, And Summerize it at order_seq level

data.beauty.top.dd10 <- data.beauty2015[ deal_duration == 10 & deal_id %in%                                                   top_deals$deal_id,
                                         list( gb = sum(gb)),by="order_seq"]
data.beauty.top.dd10 <- data.beauty.top.dd10[ order_seq >=0,]
temp.df <- data.table( order_seq = seq(min(data.beauty.top.dd10$order_seq),
                                       max(data.beauty.top.dd10$order_seq),1))
data.beauty.top.dd10 <- merge(temp.df,data.beauty.top.dd10,by="order_seq",all.x = T)
data.beauty.top.dd10$gb[ is.na(data.beauty.top.dd10$gb)] <- 0

Get predicted demand

exponential.model <- lm(log(gb)~order_seq,data = data.beauty.top.dd10[ order_seq >= 25 & order_seq <= 40,])
data.beauty.top.dd10$pre_after25 <- exp(predict(exponential.model,data.beauty.top.dd10))
exponential.model <- lm(log(gb+1)~order_seq,data = data.beauty.top.dd10[ order_seq <= 9,])
data.beauty.top.dd10$pre_before10 <- exp(predict(exponential.model,data.beauty.top.dd10))

write.csv(data.beauty.top.dd10,file = 'leiure_predicted_trend.csv',row.names = F)
ggplot(aes(x=order_seq),data=data.beauty.top.dd10) + 
  geom_line(aes(y=gb,group=1),stat='identity',color='red')+
  geom_line(aes(y=pre_before10,group=1),stat='identity',color='blue')+
  geom_line(aes(y=pre_after25,group=1),stat='identity',color='green')+
  scale_x_continuous(limits=c(0,75),breaks=seq(0,150,5))+
  scale_y_continuous(limits=c(0,20000000))+
  labs(y = 'total GB',x = 'Nth day from the day, deal starts',title="Total GB at Nth day after deal launch")
